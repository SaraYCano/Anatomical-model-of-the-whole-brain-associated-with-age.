folder = Path("fc")
files = list(folder.glob("*.txt"))

Matrix_age_correlation=[]  
triu_indices_FC = np.triu_indices(20, k=1) ### PARA GUARDAR LA PARTE DE ARRIBE SIN LA DIAGONAL PRINCIPAL 



for file in files:
    signal_bold = np.loadtxt(file) 
     #Transformada rápida de Fourier FFT en el tiempo 
    ts_fft = fft(signal_bold, axis=1) 
    # Crear máscara de banda 0.01-0.2 Hz
   
    mask_fft = (np.abs(v_f) >= 0.01) & (np.abs(v_f) <= 0.2) #definir la mascara booleana
    ts_fft[:, ~mask_fft] = 0 #aplicar e invertir invierte la mascara
    filt_ts = np.real(ifft(ts_fft, axis=1)) # inversa de fourier, la devuelve al tiempo
    
    filt_ts_T = filt_ts.T # de (20,159) a (159,20) Transponer las series 

    
    signal_bold_norm = zscore(filt_ts_T, axis=0) #normalización
    
    corr_matrix = np.corrcoef(signal_bold_norm, rowvar=False) # Correlation matrix 20x20

    ####   VECTORIZACIÓN    ######
    vectorized = corr_matrix[triu_indices_FC] #Extract upper half without diagonal
    Matrix_age_correlation.append(vectorized)
    #print(vectorized)
FC = np.array(Matrix_age_correlation)
print("FC =", FC.shape) 
#print( vectorized.shape)
